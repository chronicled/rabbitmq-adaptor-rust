// We're using Rust 2018
#![warn(rust_2018_idioms)]
// Long and nested future chains can quickly result in large generic types.
#![type_length_limit = "16777216"]

//! This library wraps some verbose and repetitive laptin-futures patterns with the future builder
//! approach.

use std::collections::HashMap;
use std::net::SocketAddr;
use std::sync::{Arc, Mutex};

use enclose::enclose;
use failure::err_msg;
use failure::Error as FailureError;
use futures::{Async, Poll, Stream, task};
use futures::future::{self, Future, Loop, ok};
use lapin_futures::channel::{
    BasicConsumeOptions, BasicProperties, BasicPublishOptions, Channel, ConfirmSelectOptions,
    ExchangeDeclareOptions, QueueBindOptions, QueueDeclareOptions,
};
use lapin_futures::client::{Client as LapinClient, ConnectionOptions, Heartbeat};
use lapin_futures::error::Error as LapinError;
use lapin_futures::message::Delivery;
use lapin_futures::queue::Queue;
use lapin_futures::types::FieldTable;
use log::{error, trace};
use log::info;
use tokio;
use tokio::net::TcpStream;
use tokio::sync::oneshot;
use uuid::Uuid;

static MAX_RETRIES: u8 = 5;

/// Struct which manages the RabbitMQ connection
///
/// # Examples
///
/// ```no_run
/// let rabbit = RabbitConnection::new();
/// let result = rabbit.connect("127.0.0.1", 8080).create_topic_exchange("foo").into_future().exec();
/// ```
#[derive(Clone)]
pub struct RabbitConnection(Arc<Mutex<RabbitConnectionInner>>);
pub struct RabbitConnectionInner {
    client: Option<LapinClient<TcpStream>>,
    default_channel: Option<Channel<TcpStream>>, // default channel for sending, lapin consumers require dedicated channels
    retry_counter: u8,
    error: bool,
    reply_queue_name: String,
    pending_replies: HashMap<String, oneshot::Sender<Delivery>>,
}

impl Default for RabbitConnection {
    fn default() -> RabbitConnection {
        RabbitConnection(Arc::new(Mutex::new(RabbitConnectionInner {
            client: None,
            default_channel: None,
            retry_counter: 0,
            error: false,
            reply_queue_name: Uuid::new_v4().to_string(),
            pending_replies: HashMap::new(),
        })))
    }
}

impl RabbitConnection {
    /// Establishes the connection
    ///
    /// # Examples
    /// ```no_run
    /// # let rabbit = RabbitConnection::new();
    /// rabbit.connect("127.0.0.1", 8080);
    /// ```
    pub fn connect(&self, host: &str, port: u16) -> () {
        let addr_str = format!("{}:{}", host, port);
        let addr = addr_str.parse::<SocketAddr>().unwrap();
        // todo: connection retries and error handling
        tokio::spawn(future::loop_fn(
            (),
            enclose!((self => con) move |_| {
                tcpstream_future(&addr)
                    .and_then(connect_future)
                    .map(heartbeat_future)
                    .inspect(enclose!((con) move |client| {
                        let mut con_lock = con.0.lock().unwrap();
                        con_lock.client.replace(client.clone());
                    }))
                    .and_then(create_channel)
                    .and_then(enclose!((con) move |channel| {
                        {
                          let mut con_lock = con.0.lock().unwrap();
                          con_lock.default_channel.replace(channel.clone());
                        }
                        setup_reply_queue(con) //blocking
                    }))
                    .then(enclose!((con) move |res| {
                        let mut con_lock = con.0.lock().unwrap();
                        match res {
                            Ok(_) => {
                                con_lock.retry_counter = 0;
                                con_lock.error = false;
                                Ok(Loop::Break(()))
                            },
                            Err(err) => {
                                con_lock.retry_counter += 1;
                                if con_lock.retry_counter == MAX_RETRIES {
                                    con_lock.error = true;
                                    println!("Connection error {:?}", err);
                                    Ok(Loop::Break(()))
                                } else {
                                    println!("Retrying connection (n = {})", con_lock.retry_counter);
                                    Ok(Loop::Continue(()))
                                }
                            },
                        }
                    }))
            }),
        ));
    }
}

impl Future for RabbitConnection {
    type Item = RabbitConnection;
    type Error = FailureError;

    fn poll(&mut self) -> Poll<Self::Item, Self::Error> {
        // the connection has the three following states
        // 1. channel ready -> resolve future
        // 2. channel pending -> wait some more time until notify
        // 3. channel error -> resolve to error
        let con = self.0.lock().unwrap();
        match &con.default_channel {
            Some(_) => Ok(Async::Ready(self.clone())),
            None => match con.error {
                false => {
                    task::current().notify(); // todo: optimize this, also occurs in lapin
                    Ok(Async::NotReady)
                }
                true => Err(err_msg("No channel to RabbitMQ available")),
            },
        }
    }
}

type RabbitFuture = Box<dyn Future<Item = RabbitConnection, Error = FailureError> + Send>;

/// RabbitExt overloads the futures generated from the connect() method to allow for easy chaining
/// of futures.
pub trait RabbitExt {
    /// Creates a topic exchange
    /// https://www.rabbitmq.com/tutorials/amqp-concepts.html#exchange-topic
    fn create_topic_exchange(self, name: &'static str) -> RabbitFuture;

    /// Creates a queue
    /// https://www.rabbitmq.com/tutorials/amqp-concepts.html#queues
    fn create_queue(self, name: String) -> RabbitFuture;

    /// Binds a queue to a topic exchange
    /// https://www.rabbitmq.com/tutorials/amqp-concepts.html#bindings
    fn bind_queue(
        self,
        name: &'static str,
        exchange: &'static str,
        routing_key: &'static str,
    ) -> RabbitFuture;

    /// Register a push consumer, supplying a callback which is executed on each incoming message
    /// The callback function should return true for ack and false for nack
    /// https://www.rabbitmq.com/tutorials/amqp-concepts.html#consumers
    fn register_consumer<F: Fn(Delivery) -> bool + Send + 'static>(
        self,
        queue_name: String,
        callback: F,
    ) -> Box<dyn Future<Item = (), Error = FailureError> + Send>;

    /// Publish a message to a topic exchange
    fn publish(self, exchange: &'static str, topic: &'static str, payload: Vec<u8>)
        -> RabbitFuture;

    /// Publish a RPC message to a topic exchange
    fn rpc_publish(
        self,
        exchange: &'static str,
        topic: &'static str,
        payload: Vec<u8>,
    ) -> Box<dyn Future<Item = Delivery, Error = FailureError> + Send>;

    fn rpc_reply(self, message: &Delivery, payload: Vec<u8>) -> RabbitFuture;

    /// Close the channel gracefully
    fn close_channel(self) -> Box<dyn Future<Item = (), Error = FailureError> + Send>;
}

impl<T: Future<Item = RabbitConnection, Error = FailureError> + Send + 'static> RabbitExt for T {
    fn create_topic_exchange(self, name: &'static str) -> RabbitFuture {
        Box::new(self.and_then(move |con: RabbitConnection| {
            let channel = { con.0.lock().unwrap().default_channel.clone() };
            channel
                .unwrap()
                .exchange_declare(
                    name,
                    "topic",
                    ExchangeDeclareOptions {
                        durable: true,
                        ..Default::default()
                    },
                    FieldTable::new(),
                )
                .map(enclose!((con) move |_| con))
                .map_err(FailureError::from)
        }))
    }

    fn create_queue(self, name: String) -> RabbitFuture {
        // because of a bug in lapin, we need to open a temporary channel for this operation
        Box::new(self.and_then(move |con: RabbitConnection| {
            let channel = { con.0.lock().unwrap().default_channel.clone() };
            channel
                .unwrap()
                .queue_declare(
                    &name,
                    QueueDeclareOptions {
                        durable: true,
                        ..Default::default()
                    },
                    FieldTable::new(),
                )
                .map(enclose!((con) move |_| con))
                .map_err(FailureError::from)
        }))
    }

    fn bind_queue(
        self,
        name: &'static str,
        exchange: &'static str,
        routing_key: &'static str,
    ) -> RabbitFuture {
        Box::new(self.and_then(move |con: RabbitConnection| {
            let client = { con.0.lock().unwrap().client.clone() };
            client
                .unwrap()
                .create_confirm_channel(ConfirmSelectOptions::default())
                .and_then(move |channel| {
                    channel
                        .queue_bind(
                            name,
                            exchange,
                            routing_key,
                            QueueBindOptions {
                                ..Default::default()
                            },
                            FieldTable::new(),
                        )
                        .and_then(enclose!((channel) move |_| channel.close(200, "Bye")))
                })
                .map(enclose!((con) move |_| con))
                .map_err(FailureError::from)
        }))
    }

    // Blocks a dedicated channel and registers a consumer.
    // The callback will be called for incoming messages.
    // If the callback returns true, basic.ack will be sent in response.
    // If the callback returns false, basic.nack will be sent in response.
    fn register_consumer<F: Fn(Delivery) -> bool + Send + 'static>(
        self,
        queue_name: String,
        callback: F,
    ) -> Box<dyn Future<Item = (), Error = FailureError> + Send> {
        Box::new(
            self.and_then(move |con: RabbitConnection| {
                let client = {con.0.lock().unwrap().client.clone()};
                client
                    .unwrap()
                    .create_confirm_channel(ConfirmSelectOptions::default())
                    .and_then(|channel| {
                        // TODO: Due to a bug in lapin-async, we need to send a queue declaration to
                        // the server on the same channel. Otherwise some internal consumer data is not
                        // initialized (see channels[id].queues.consumers in api.rs).
                        channel
                            .queue_declare(
                                &queue_name,
                                QueueDeclareOptions {
                                    durable: true,
                                    ..Default::default()
                                },
                                FieldTable::new(),
                            )
                            .and_then(enclose!((channel) move |_| {
                                channel.basic_consume(
                                    &Queue::new(queue_name, 0, 0),
                                    &Uuid::new_v4().to_string(),
                                    BasicConsumeOptions::default(),
                                    FieldTable::new(),
                                )
                            }))
                            .and_then(enclose!((channel) move |stream| {
                                stream
                                    .for_each(enclose!((channel) move |message| {
                                        match callback(message.clone()) {
                                            true => tokio::spawn(channel.basic_ack(message.delivery_tag, false).map_err(|_| ())),
                                            // when sending nack, we don't have a use case for requeuing yet
                                            false => tokio::spawn(channel.basic_nack(message.delivery_tag, false, false).map_err(|_| ())),
                                        };
                                        ok::<(), LapinError>(())
                                    }))
                            }))
                    })
                    .map_err(FailureError::from)
            })
            .map_err(move |err| {
                error!("Error registering consumer {:?}", err);
                FailureError::from(err)
            })
        )
    }

    fn publish(
        self,
        exchange: &'static str,
        topic: &'static str,
        payload: Vec<u8>,
    ) -> RabbitFuture {
        Box::new(self.and_then(move |con: RabbitConnection| {
            let channel = { con.0.lock().unwrap().default_channel.clone() };
            channel
                .unwrap()
                .basic_publish(
                    exchange,
                    topic,
                    payload,
                    BasicPublishOptions::default(),
                    BasicProperties::default(),
                )
                .map(enclose!((con) move |_| con))
                .map_err(FailureError::from)
        }))
    }

    fn rpc_publish(
        self,
        exchange: &'static str,
        topic: &'static str,
        payload: Vec<u8>,
    ) -> Box<dyn Future<Item = Delivery, Error = FailureError> + Send> {
        let (sender, receiver) = oneshot::channel::<Delivery>();
        let correlation_id = Uuid::new_v4().to_string();
        tokio::spawn(
            self.and_then(enclose!((correlation_id) move |con: RabbitConnection| {
                let (channel, reply_queue_name) = {
                    let mut con_lock = con.0.lock().unwrap();
                    con_lock.pending_replies.insert(correlation_id.clone(), sender);
                    (con_lock.default_channel.clone(), con_lock.reply_queue_name.clone())
                };
                channel
                    .unwrap()
                    .basic_publish(
                        exchange,
                        topic,
                        payload,
                        BasicPublishOptions::default(),
                        BasicProperties::default()
                            .with_correlation_id(correlation_id)
                            .with_reply_to(reply_queue_name)
                    )
                    .map_err(FailureError::from)
            }))
            .map(|_| ())
            .map_err(|err| error!("{:?}", err)),
        );
        Box::new(receiver.map_err(FailureError::from))
    }

    fn rpc_reply(self, message: &Delivery, payload: Vec<u8>) -> RabbitFuture {
        match (
            message.properties.reply_to().clone(),
            message.properties.correlation_id().clone(),
        ) {
            (Some(reply_to), Some(correlation_id)) => {
                Box::new(self.and_then(move |con: RabbitConnection| {
                    let channel = { con.0.lock().unwrap().default_channel.clone() };
                    channel
                        .unwrap()
                        .basic_publish(
                            "",
                            &reply_to,
                            payload,
                            BasicPublishOptions::default(),
                            BasicProperties::default().with_correlation_id(correlation_id),
                        )
                        .map(enclose!((con) move |_| con))
                        .map_err(FailureError::from)
                }))
            }
            _ => {
                error!(
                    "Cannot send RPC reply for a message without reply_to or correlation_id field."
                );
                Box::new(self)
            }
        }
    }

    fn close_channel(self) -> Box<dyn Future<Item = (), Error = FailureError> + Send> {
        Box::new(self.and_then(move |con: RabbitConnection| {
            let channel = { con.0.lock().unwrap().default_channel.clone() };
            channel
                .unwrap()
                .close(200, "Bye")
                .map_err(FailureError::from)
                .map(|_| ())
        }))
    }
}

fn heartbeat_future(
    (client, heartbeat): (
        LapinClient<TcpStream>,
        Heartbeat<impl Future<Item = (), Error = LapinError> + Send + 'static>,
    ),
) -> LapinClient<TcpStream> {
    tokio::spawn(heartbeat.map_err(|e| info!("heartbeat error: {}", e)));
    client
}

fn tcpstream_future(addr: &SocketAddr) -> impl Future<Item = TcpStream, Error = FailureError> {
    TcpStream::connect(&addr).map_err(FailureError::from)
}

fn connect_future(
    stream: TcpStream,
) -> impl Future<
    Item = (
        LapinClient<TcpStream>,
        Heartbeat<impl Future<Item = (), Error = LapinError> + Send + 'static>,
    ),
    Error = FailureError,
> {
    LapinClient::connect(
        stream,
        ConnectionOptions {
            frame_max: 65535,
            ..Default::default()
        },
    )
    .map_err(FailureError::from)
}

fn create_channel(
    client: LapinClient<TcpStream>,
) -> impl Future<Item = Channel<TcpStream>, Error = FailureError> {
    client
        .create_confirm_channel(ConfirmSelectOptions::default())
        .map_err(FailureError::from)
}

fn reply_queue_callback(con: RabbitConnection, message: Delivery) -> bool {
    match message.properties.correlation_id() {
        Some(correlation_id) => {
            let sender = { con.0.lock().unwrap().pending_replies.remove(correlation_id) };
            match sender {
                Some(sender) => {
                    sender.send(message.clone()).unwrap();
                    trace!("Sending {:?}", message);
                }
                None => error!("Ignoring message with unknown correlation_id on reply queue"),
            }
        }
        None => error!("Ignoring message without correlation_id on reply queue"),
    };
    true
}

fn setup_reply_queue(con: RabbitConnection) -> impl Future<Item = (), Error = FailureError> {
    let reply_queue_name = con.0.lock().unwrap().reply_queue_name.clone();
    con.clone().register_consumer(
        reply_queue_name,
        enclose!((con) move |message| {
          return reply_queue_callback(con.clone(), message);
        }),
    )
}
